187 inline void graduationChairPos(int index, float& x, float& y) {
188 const int row = index / GRAD_CHAIR_COLS;
189 const int col = index % GRAD_CHAIR_COLS;
190 const float gridW = GRAD_CHAIR_COLS * GRAD_CHAIR_W
191 + (GRAD_CHAIR_COLS - 1) * GRAD_CHAIR_COL_GAP;
192 const float startX = (WORLD_W - gridW) * 0.5f + GRAD_CHAIR_W * 0.5f;
193 x = startX + col * (GRAD_CHAIR_W + GRAD_CHAIR_COL_GAP);
194 y = graduationChairZoneBottomY() - row * graduationRowStep();
195 }

232 inline int graduationNearestChairInRow(int row, float worldX) {
233 int best = graduationPathChairIndex(row);
234 float bestDist = 1e9f;
235 for (int col = 0; col < GRAD_CHAIR_COLS; ++col) {
236 const int index = row * GRAD_CHAIR_COLS + col;
237 float x = 0.0f, y = 0.0f;
238 graduationChairPos(index, x, y);
239 const float dist = x > worldX ? x - worldX : worldX - x;
240 if (dist < bestDist) {
241 bestDist = dist;
242 best = index;
243 }
244 }
245 return best;
246 }
